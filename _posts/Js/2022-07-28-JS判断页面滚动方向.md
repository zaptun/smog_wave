### 方法

```javascript
let previousScrollPosition = 0;

const isScrollingDown = () => {
    let goingDown = false;

    let scrollPosition = window.pageYOffset;

    if (scrollPosition > previousScrollPosition) {
        goingDown = true;
    }

    previousScrollPosition = scrollPosition;

    return goingDown;
};
```
