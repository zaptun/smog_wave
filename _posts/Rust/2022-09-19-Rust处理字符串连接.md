```rust
let first = "First";
let second = "Second";

//1. use join method
fn main() {
    let first = "First";
    let second = "Second";
    let output = [first, second].join(" ");
    print!("{}", output);
}
//Output: First Second

// 2. use format macro
fn main() {
    let first = "First";
    let second = "Second";
    let output = format!("{} {}", first, second);
    print!("{}", output);
}
//Output: First Second

// 3. use concat macro concat
fn main() {
    let output = concat!("First", " ", "Second");
    print!("{}", output);
}
//Output: First Second

// 4. push_str method
fn main() {
    let mut first = "First".to_string();
    let second = "Second".to_string();
    first.push_str(&second);
    println!("{}", first);
}
//Output: FirstSecond

// 5. use + operator
fn main() {
    let first = "First".to_string();
    let second = "Second".to_string();
    let result = first + &second;
    println!("{}", result);
}
//Output: FirstSecond

```
